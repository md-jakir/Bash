#!/bin/bash
#find all directories and files from a directory.
dirpath=/root/test/
d_array() {
	cd $dirpath; dir=(*)
	c=0; max=4
	while [ $c -le $max ]; do 
		if [ -d ${dir[$c]} ]; then
			echo ${dir[$c]}
		fi
		c=$[$c+1]
	done	
}
f_array() {
	cd $dirpath; file=(*)
	m=0; max1=4
	while [ $m -le $max1 ]; do
		if [ -f ${file[$m]} ]; then
			echo ${file[$m]}
		fi
		m=$[$m+1]
	done
}
check() {
	d_array
	f_array	
	
}
check
